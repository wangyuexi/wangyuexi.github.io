int IsPath(AdjList g,int vi,int vj){
    Queue Q;
    visited[vi] = true;
    InitQueue(&Q);
    EnterQueue(&Q,vi);
    while (!IsEmpty(Q)){
        DeleteQueue(&Q,&v);
        p = g.vertexesTable[v].firstarc;
        while (p != NULL){
            if(!visited[p -> adjvex]){
                if (vj == p -> adjvex)
                    return true;
                visited[p -> adjvex] = true;
                EnterQueue(&Q,p -> adjvex);
            }
            p = p -> nextarc;
        }
    }
    return false;
}

void BFS(Graph G,int v){
    visit[v];
    visited[v] = true;
    EnQueue(Q,v);
    while (!IsEmpty(Q)){
        DeQueue(Q,v);
        for (w = FirstNeighbor(G,v);w >= 0;w = NextNeighbor(G,v,w)){
            if (!visit[w])
                visit[w];
            	visited[w] = true;
            	EnQueue(Q,w);
        }
    }
}
