BiTNode *FirstNodeLRD(BiTree bt){
    p = bt;
    while (p != NULL && (p -> LChild != NULL || p -> RChild != NULL)){
        if (p -> LChild != NULL)
            p = p -> LChild;
        else
            p = p -> RChild;
    }
    return p;
}

算法三(2001)—over

计算二叉树的高度

//先序遍历求二叉树的高度(递归实现)
int PostTreeDepth(BiTree bt){
    int hl,hr,max;
    if (bt != NULL){
        hl = PostTreeDepth(bt -> LChild);	//求左子树的深度
        hr = PostTreeDepth(bt -> RChild);	//求右子树的高度
        max = hl > hr ? hl : hr;			//得到左右子树中深度较大的一个
        return (max + 1);					//返回树的深度
    }
    else
        return 0;
}

//后序遍历求二叉树的高度(递归实现)
void PreTreeDepth(BiTree bt,int h){
    if (bt != NULL){
        if (h > depth)
            depth = h;
     	PreTreeDepth(bt -> LChild,h + 1);
        PreTreeDepth(bt -> RChild,h + 1);
	}
}

算法四(2001,2003)—–以前呆逼，现在会了

计算树的最大宽度

typedef struct{
    BiTree data[MaxSize];					//保存队列中的结点指针
    int level[MaxSize];						//保存data中相同下标结点的层次
    int front,rear;
}Qu;

int BTWidth(BiTree b){
    BiTree p;
    int k,max,i,n;
    Qu.front = Qu.rear = -1;				//队列为空
    Qu.rear++;
    Qu.data[Qu.rear] = b;					//根结点指针入队
    Qu.level[Qu.rear] = 1;					//根结点层次为1
    while (Qu.front < Qu.rear){
        Qu.front++;							//出队
        p = Qu.data[Qu.front];				//出队的结点
        k = Qu.level[Qu.front];				//出队结点的层次
        if (p -> lchild != NULL){			//左孩子进队列
            Qu.rear++;
            Qu.data[Qu.rear] = p -> lchild;
            Qu.level[Qu.rear] = k + 1;
        }
        if (p -> rchild != NULL){			//右孩子进队列
            Qu.rear++;
            Qu.data[Qu.rear] = p -> rchild;
            Qu.level[Qu.rear] = k + 1;
        }
    }
    max = 0;i = 0;							//max保存同一层最多的结点个数
    k = 1;									//k表示从第一层开始查找
    while (i <= Qu.rear){					//i扫描队中所有元素
        n = 0;								//n统计第k层的结点个数
        while (i < Qu.rear && Qu.level[i] == k){
            n++;
            i++;
        }
        k = Qu.level[i];
        if (n > max)
            max = n;						//保存最大的n
    }
    return max;
}

算法五(2001,2003)

对二叉树结点进行编号，从1开始，要求每一个结点的编号大于其左右孩子，同一个结点的左右孩子中，左孩子的编号小于右孩子的编号，请问使用什么次序的遍历方式实现编号，写出非递归算法

void PostOrder(BiTree T){
    InitStack(S);
    p = T;
    r = NULL;
    k = 1;
    while (p || !IsEmpty(S)){
        if (p){									//走到最左边
            Push(S,p);
            p = p -> lchild;
        }
        else{									//向右
            GetTop(S,p);						//读栈顶结点(非出栈)
            if (p -> rchild && p ->rchild != r)	//若右子树存在，且未被访问过
                p = p -> rchild;				//转向右
            else								//否则，弹出结点并访问
                pop(S,p);						//将结点弹出
            	p -> number = k;				//赋值
            	k++;						
            	r = p;							//记录最近访问过的结点
            	p = NULL;						//结点访问完后，重置p指针
        }
    }
}

BiTNode *InNext(BiTNode * p){

        Next = p -> RChild;

二叉排序树采用二叉链的方式存放，树中的结点值各不相同，欲得到一个从大到小的递减序列，使用非递归算法。

//写个中序遍历的非递归算法，压压惊
void InOrder(BiTree T){
    InitStack(S);
    BiTree p = T;
    while (p || !IsEmpty(S)){
        if(p)						//一路向左
            push(S,p);
        	p = p -> lchild;		//左孩子不空一直向左走
        else
            pop(S,p);
        	visit(p);
        	p = p -> rchild;
    }
}

算法八(2002,2003,2005,2010,2018)—我说我没看懂，信吗？

一棵树采用孩子-兄弟的方式存放，结点结构为：

fch data nsib level

其中fch表示指向第一个孩子，nisib表示指向的下一个兄弟，level表示结点层次，设根节点层次置为1，其他以此类推，编写一算法，将树中所有结点层次值置入相应level域。并要求由跟开始逐层输出树的各条边。输出形式（AB,AC……），求树的高度。

void Traverse(BiTree T){
    TNode *p = T, *q = T;
    int level = 1;
    T -> level = level;
    if (T != NULL){
        EnQueue(&Q,p);
        while (!IsEmpty(Q)){
            DeQueue(&Q,&p);
            level = p -> level;
            for (q = p -> fch;q != NULL;q = q -> nsib){
                print(“(%c,%c)”,p->data,q->data);
         		q->level= level+1;
				EnQueue(&Q,&p);
            }
        }
    }
}
