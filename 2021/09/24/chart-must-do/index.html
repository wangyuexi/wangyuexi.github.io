算法一(2001,2004)—这T喵的能记住？

求出无向连通图中距离V0的最短路径的长度（最短路径以边数为单位计算）为K的所有的结点，要求尽可能地节省时间

void bfsKLevel(Graph g,int v0,int K){
	InitQueue(Q1);							//Q1是存放已访问顶点的队列
    InitQueue(Q2);							//Q2是存放已访问顶点层号的队列
    for (i = 0;i < g.vexnum;i++)
        visited[i] = false;					//初始化访问标志数组
    visited[v0] = true;
    level = 1;
    EnterQueue(Q1,v0);
    EnterQueue(Q2,level);
    while (!IsEmpty(Q1) && level < k + 1){
        v = DeleteQueue(Q1);				//取出已访问的顶点号
        level = DeleteQueue(Q2);			//取出已访问的层
        w = FirstAdjVertex(g,v);			//找到第一个邻接点
        while (w != -1){
            if (!visited[w]){
                visited[w] = true;
                //v的level为K,表示w的level为K+1,符合要求，输出，该结点无需进队列
                if (level == k)
                    print("%d",w);
                else
                {
                    EnterQueue(Q1,w);
                    EnterQueue(Q2,level + 1);
                }
                w = NextAdjVertex(g,v,w);	//找下一个邻接点
            }
        }
    }
}

算法二(2006)

已知连通无向图采用邻接表方式存放，求从一点Vi到另一点Vj（Vi!=Vj）所经结点数目最少的路径

void shortpath(AdjGraph *g,int u,int v){
    ArcNode *p;
    int w,i;
    Queue qu[MAX];
    int front = -1,rear = -1;
    int visited[MAX];
    for (i = 0;i < g.vexnum;i++)
        visited[i] = 0;
    rear++;
    qu[rear].data = u;
    qu[rear].parent = -1;
    visited[u] = 1;
    while (front <= rear){
        front++;
        w = qu[front].data;
        if (w == v){
            i = front;
            while (qu[i].parent != -1){
                print("%d",qu[i].data);
                i = qu[i].parent;
            }
            print("%d\n",qu[i].data);
            break;
        }
        p = g -> adjlist[w].firstarc;
        while (p != NULL){
            if (visited[p -> adjvex] == 0){
                visited[p -> adjvex] = 1;
                rear++;
                qu[rear].data = p -> adjvex;
                [rear].parent = front;
            }
            p = p -> nextarc;
        }
    }
}

int IsPath(AdjList g,int vi,int vj){
    Queue Q;
    visited[vi] = true;
    InitQueue(&Q);
    EnterQueue(&Q,vi);
    while (!IsEmpty(Q)){
        DeleteQueue(&Q,&v);
        p = g.vertexesTable[v].firstarc;
        while (p != NULL){
            if(!visited[p -> adjvex]){
                if (vj == p -> adjvex)
                    return true;
                visited[p -> adjvex] = true;
                EnterQueue(&Q,p -> adjvex);
            }
            p = p -> nextarc;
        }
    }
    return false;
}

算法三(2012)—还行还行

王道书上，广度优先遍历算法

bool visited[MAX_VERTEX_NUM];
void BFSTraverse(Graph G){
    for (i = 0;i < G.vexnum;i++)
        visited[i] = false;
    InitQueue(Q);
    for (i = 0;i < G.vexnum;i++)
        if (!visited[i])
            BFS(G,i);
}
void BFS(Graph G,int v){
    visit[v];
    visited[v] = true;
    EnQueue(Q,v);
    while (!IsEmpty(Q)){
        DeQueue(Q,v);
        for (w = FirstNeighbor(G,v);w >= 0;w = NextNeighbor(G,v,w)){
            if (!visit[w])
                visit[w];
            	visited[w] = true;
            	EnQueue(Q,w);
        }
    }
}

算法五(2016)—真T喵的长

某有向图采用邻接表存储，编写算法输出该图的拓扑结构

void FindID(AdjList G,int indegree[MAX_VERTEX_NUM]){
    int i;
    ArcNode *p;
    for (i = 0;i < G.vexnum;i++)
        indegree[i] = 0;
    for (i = 0;i < G.vexnum;i++){
        p = G.vertex[i].firstarc;
        while (p != NULL){
            indegree[p -> agjvex]++;
            p -> nextarc;
        }
    }
}

int TopoSort(AdjList G){
    Stack s;
    int indegree[MAX_VERTEX_NUM]
    int i,count,k;
    ArcNode *p;
    FindID(G,indegree);
    InitStack(&s);
    for (i = 0;i < G.vexnum;i++){
        if (indegree[i] == 0)
            push(&s,i);
    }
    count = 0;
    while (!IsEmpty(s)){
        pop(&s,&i);
        print("%c",G.vertex[i].data);
        count++;
        p = G.vertex[i].firstarc;
        while (p != NULL){
            k = p -> adjvex;
            indegree[k]--;
            if (indegree[k] == 0)
                push(&s,k);
            p = p -> nextarc;
        }
        if (count < G.vexnum)
            return error;
        else
            return ok;
    }
}
